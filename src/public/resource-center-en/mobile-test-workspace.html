// Enable mobile testing tools by environment and role
const tools = MobileTestWorkspace.init({
  env: "test",
  appId: "business-mobile-app",
  enable: ["postman", "vconsole", "busApi", "observability"],
  auth: {
    tenantId: "tenant-demo",
    tokenProvider: () => getAccessToken()
  },
  guard: {
    allowRoles: ["developer", "tester", "ops"],
    productionReadonly: true
  }
});

tools.open();
